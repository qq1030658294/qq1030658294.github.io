基于自适应算法下的智能交通灯

我只负责硬件部分所以对硬件部分进行一个回顾：
1、 设计思路

　　基于这个自适应算法，要得到模拟和应用，要解决的首要问题是单片机一个联网问题，怎么解决？esp8266模块！这个模块可以实现单片机和服务器的数据交互。服务器通过UDP协议发送数据给到esp8266，再由8266的串口发到单片机上，发的数据类型是ascll码，单片机通过对发送过来的数据进行处理，怎么处理？这个是esp8266发过来的数据格式（+IPD,0,11:43 32 10 10,0, CLOSED）我们只需要用到43（东西方向直行时间）,32（东西方向左转时间）,10（南北方向直行时间）,10（东西方向左转时间），这个是数字是ascll码，需要对其进行处理，1对应的ascll为49，所以要使用1的话就要使其减去48，其他的数字同理，但是我们运用的数字是两位数，所以得到的1得要乘以10,，再加上个数数字，同理其他的三个数也因如此。这样我们就分别可以得到43 32 10 10，这四个数。得到了这个四个数，我们就得把应用到自己通过单片机设计的交通灯上。那么这个交通灯的逻辑和收发数据是怎么实现的呢？

　　那就是通过C语言，在keil里编程，编程问题首要解决的问题是交通灯逻辑，我们的设计先让东西方向直行先行，东西方向左再行，在还有最后3秒时让其处于一个黄灯的状态，让南北方向处于一个等待，之后再交换。当我们数据传过来了之后，再下一个周期前5s,对数据进行更换。

　　那么这个数据是怎么拿到的？单片机串口P3.0和P3.1，这个两个串口是单片机用来跟外部模块进行传数据用的，当我们把波特率（9600）和相关的开关设置好了之后，再把esp8266的TX,RX用导线连接好了之后，就可以使用正常的数据收发了。接下来就是数据处理，上面已经说过了，这里不再赘述。接着就是数据应用，我们的交通灯是通过值传递来应用的，所以只需将上文的数据传给对应的数组就能应用上了。

　　还有最后一个问题，怎么让应用上的数据实现减一功能？定时器，和上面的串口一样，打开相应的寄存器，再赋初值，这样就能实现1s的定时中断，在中断函数里面，使对应的值减一（每一秒时进入中断，使对应的值减一），这样就能实现倒计时功能。

　　以上就是交通灯设计思路。具体的实现放在相应的文档里面。

/* 触摸屏实现 */
int Get_Touch_Motion()
{    
    struct input_event ts;

    start.x = start.y = stop.x = stop.y = -1;//在坐标范围之外，-1,-1读不到

    int temp_x, temp_y;
    while(1)
    {
        read(ts_fd, &ts, sizeof(ts));
        if(ts.type == EV_ABS)//判断是否是绝对事件
        {
            if(ts.code==ABS_X)
            {
                // temp_x = ts.value*800/1024;
                temp_x = ts.value;
            }
            else if(ts.code == ABS_Y)
            {
                // temp_y = ts.value*480/600;    
                temp_y = ts.value;    
            }
#if 1 //    非零为真，为假不编译，此处是蓝屏使用            
            if(ts.code == ABS_PRESSURE)
            {
                printf("ABS_PRESSURE = %d\n", ts.value);
                if(ts.value > 0)
                {
                    if(start.x<0 && start.y<0)
                    {
                        start.x = temp_x;
                        start.y = temp_y;
                    }
                    printf("按下%d\n", ts.value);
                }
                else
                {
                    stop.x = temp_x;
                    stop.y = temp_y;
                    printf("松开\n");
                    break;
                }
            }
#endif            
        }
#if 0    //非零为真，为假不编译，此处是黑屏使用    
        else if(ts.type == EV_KEY && ts.code == BTN_TOUCH)//&& buf.value == 0
        {
            if(ts.value > 0)//记录第一次的坐标（起始）
            {
                if(start.x<0 && start.y<0)//判断第一次进来
                {
                    start.x = temp_x;
                    start.y = temp_y;
                }
                printf("按下%d\n", ts.value);
            }
            else//记录最后一次坐标（终点）
            {
                stop.x = temp_x;
                stop.y = temp_y;
                printf("松开\n");
                break;
            }
        }
#endif        
    }
    printf("start(%d,%d), stop(%d,%d)\n", start.x, start.y, stop.x, stop.y);

    int x, y;//接收终点减去起点的差值
    x = stop.x - start.x; 
    y = stop.y - start.y;
    printf("X:%d, Y:%d\n", x, y);

    if(x*x + y*y > 40*40)//小于40表示点击
    {
        if(abs(y) > abs(x) && y<0)
            return 1;//up
        else if(y > abs(x) && y>0)
            return 2;//down
        else if(abs(y) < abs(x) && x<0)
            return 3;//left
        else if(x > abs(y) && x>0)
            return 4;//right
    }
    else
    {
        return 0;//touch
    }

}

/* 图片显示 */
void show_any_bmp(char *bmpname, unsigned int x0, unsigned int y0)
{
    int bmp_fd;//图片文件描述符
    char bmp_buf[800*480*3] = {0};//图片颜色存储的缓冲区
    int lcd_buf[800*480] = {0};//24转32位缓冲区
    char file_buf[54] = {0};//头信息存储
    unsigned int x, y, color, width, height;

    bmp_fd = open(bmpname, O_RDONLY);
    if(bmp_fd == -1)
    {
        perror("open() BMP failed");
        exit(1);//安全退出
    }

    //读取文件头信息
    read(bmp_fd, file_buf, 54);

    width = file_buf[18]<<0 | file_buf[19]<<8 | file_buf[20]<<16 | file_buf[21]<<24;
    height = file_buf[22]<<0 | file_buf[23]<<8 | file_buf[24]<<16 | file_buf[25]<<24;
    printf("width=%d, height=%d\n", width, height);
    if(width+x0 > 800 || height+y0 > 480)
    {
        printf("超出屏幕最大显示范围\n");
        exit(1);
    }

    //都图片的颜色数据
    // read(bmp_fd, bmp_buf, 800*480*3);
    read(bmp_fd, bmp_buf, width*height*3);


    //显示每一个像素点
    int j = 0; //跳到下一个像素点
    unsigned char b, g, r;

    for(y = y0; y < height+y0; y++)
    {        
        for(x = x0; x < width+x0; x++, j+=3)
        {
            b = bmp_buf[j];
            g = bmp_buf[j+1];
            r = bmp_buf[j+2];
            color = b | g<<8 | r<<16;
            lcd_one_pixle_point(x, 479-y, color);
        }
        usleep(1000);
    }

    close(bmp_fd);
}


//获取jpg文件的大小
unsigned long file_size_get(const char *pfile_path)
{
    unsigned long filesize = -1;    
    struct stat statbuff;

    if(stat(pfile_path, &statbuff) < 0)
    {
        return filesize;   //获取文件属性失败
    }
    else
    {
        filesize = statbuff.st_size;//获取到文件大小
    }
// printf("line:%d\n", __LINE__);
    return filesize;
}

//显示正常jpg图片
int lcd_draw_jpg(unsigned int x,unsigned int y,const char *pjpg_path)  
{
    // printf("line:%d\n", __LINE__);
    /*定义解码对象，错误处理对象*/
    struct     jpeg_decompress_struct     cinfo;
    struct     jpeg_error_mgr             jerr;    

    unsigned char     *pcolor_buf = g_color_buf;//*（pcolor_buf+0）== g_color_buf[0]
    char     *pjpg;

    unsigned int     i=0;
    unsigned int    color =0;
    //unsigned int    count =0;

    unsigned int     x_s = x;//x的映射起始位置
    unsigned int     x_e ;    
    unsigned int     y_e ;
    unsigned int    y_n    = y;//x和y的映射的起始位置
    unsigned int    x_n    = x;

             int    jpg_fd;
    unsigned int     jpg_size;

    if(pjpg_path!=NULL)//如果传进来的是空字符串，就退出
    {
        /* 申请jpg资源，权限可读可写 */    
        jpg_fd=open(pjpg_path,O_RDWR);

        if(jpg_fd == -1)
        {
           printf("open %s error\n",pjpg_path);

           return -1;    
        }    

        /* 获取jpg文件的大小 */
        jpg_size=file_size_get(pjpg_path);    
        if(jpg_size<3000)
            return -1;

        /* 为jpg文件申请内存空间 */    
        pjpg = malloc(jpg_size);
// printf("line:%d\n", __LINE__);
        /* 读取jpg文件所有内容到内存 */        
        read(jpg_fd,pjpg,jpg_size);
    }
    else
        return -1;

    /*注册出错处理*/
    cinfo.err = jpeg_std_error(&jerr);

    /*创建解码*/
    jpeg_create_decompress(&cinfo);

    /*直接解码内存数据*/        
    jpeg_mem_src(&cinfo,pjpg,jpg_size);

    /*读文件头*/
    jpeg_read_header(&cinfo, TRUE);

    /*开始解码*/
    jpeg_start_decompress(&cinfo);    


    x_e    = x_s +cinfo.output_width;
    y_e    = y  +cinfo.output_height;    

    /*读解码数据*/
    while(cinfo.output_scanline < cinfo.output_height )
    {        
        pcolor_buf = g_color_buf;
// printf("line:%d\n", __LINE__);        
        /* 读取jpg一行的rgb值 */
        jpeg_read_scanlines(&cinfo,&pcolor_buf,1);

        for(i=0; i<cinfo.output_width; i++)
        {        
            /* 获取rgb值 */
            color =         *(pcolor_buf+2);
            color = color | *(pcolor_buf+1)<<8;
            color = color | *(pcolor_buf)<<16;    
// printf("line:%d\n", __LINE__);            
            /* 显示像素点 */
            lcd_one_pixle_point(x_n,y_n,color);
// printf("line:%d\n", __LINE__);            
            pcolor_buf +=3;

            x_n++;
        }

        /* 换行 */
        y_n++;            

        x_n = x_s;
// printf("line:%d\n", __LINE__);        
    }        

    /*解码完成*/
    jpeg_finish_decompress(&cinfo);
    jpeg_destroy_decompress(&cinfo);

    if(pjpg_path!=NULL)
    {
        /* 关闭jpg文件 */
        close(jpg_fd);    
// printf("line:%d\n", __LINE__);        
        /* 释放jpg文件内存空间 */
        free(pjpg);        
    }
// printf("line:%d\n", __LINE__);
    return 0;
}

/* 摄像头 */
void *camera(void *arg)
{

    int cam_fd,jpg_fd;
    int ret, jpg_size;
    int x,y,i=0;
    char pic_name[10];
    char *jpg_mem;



    // 获取摄像头操作许可
    cam_fd = open("/dev/video7",O_RDWR);
    if(cam_fd < 0)
        {
        printf(" open  camera Failed !\n");
        // return -1; //返回异常            
        }


    struct v4l2_format fmt;

    //获得摄像头的功能参数
    struct v4l2_capability cap;
    if (-1 == ioctl (cam_fd, VIDIOC_QUERYCAP, &cap))
        {
        printf("  camera  VIDIOC_QUERYCAP  Failed !\n");
        // return -1; //返回异常            
        }

    printf("  camera  VIDIOC_QUERYCAP version : %x \n",cap.version);


    //设置摄像头的通道
    int index;
    index = 0;
    if (-1 == ioctl (cam_fd, VIDIOC_S_INPUT, &index)) {
        printf("  camera  VIDIOC_S_INPUT  Failed !\n");
        // return -1; //返回异常    
    }



    memset(&fmt,0,sizeof(fmt));
    fmt.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;

    //获取摄像头当前的采集格式get
    if (-1 == ioctl (cam_fd, VIDIOC_G_FMT, &fmt))
        {
        printf("  camera  VIDIOC_G_FMT  Failed !\n");
        // return -1; //返回异常    
        }

    printf("  camera  VIDIOC_G_FMT width : %d \n",fmt.fmt.pix.width);
    printf("  camera  VIDIOC_G_FMT height : %d \n",fmt.fmt.pix.height );

    if( V4L2_PIX_FMT_JPEG == fmt.fmt.pix.pixelformat )
    printf("  camera  VIDIOC_G_FMT pixelformat : V4L2_PIX_FMT_JPEG \n");



    //配置摄像头的采集格式
    memset(&fmt,0,sizeof(fmt));
    fmt.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
    fmt.fmt.pix.width = 640;
    fmt.fmt.pix.height = 480;
    fmt.fmt.pix.pixelformat = V4L2_PIX_FMT_JPEG;
    //fmt.fmt.pix.field = V4L2_FIELD_INTERLACED;

    if (-1 == ioctl (cam_fd, VIDIOC_S_FMT, &fmt))//set
        {
        printf("  camera  VIDIOC_S_FMT  Failed !\n");
        // return -1; //返回异常    
        }

    //再次获取摄像头当前的采集格式
    memset(&fmt,0,sizeof(fmt));
    fmt.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
    if (-1 == ioctl (cam_fd, VIDIOC_G_FMT, &fmt))
        {
        printf("  final camera  VIDIOC_G_FMT  Failed !\n");
        // return -1; //返回异常    
        }

    printf("  final camera  VIDIOC_G_FMT width : %d \n",fmt.fmt.pix.width);
    printf("  final camera  VIDIOC_G_FMT height : %d \n",fmt.fmt.pix.height );

    if( V4L2_PIX_FMT_JPEG == fmt.fmt.pix.pixelformat )
    printf("  final camera  VIDIOC_G_FMT pixelformat : V4L2_PIX_FMT_JPEG \n");



    //申请摄像头的缓存
    struct v4l2_requestbuffers reqbuf;


    memset (&reqbuf, 0, sizeof (reqbuf));

    reqbuf.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
    reqbuf.memory = V4L2_MEMORY_MMAP;
    reqbuf.count = 4;

    //申请分配缓冲块
    if (-1 == ioctl (cam_fd, VIDIOC_REQBUFS, &reqbuf)) {

        printf ("Video capturing or mmap-streaming is not supported\n");
        // return -1;
    }

    //分配四个缓冲块内存地址
    buffers = calloc (reqbuf.count, sizeof (*buffers));//malloc
    struct v4l2_buffer buffer;

    for (i = 0; i < reqbuf.count; i++) {


        memset (&buffer, 0, sizeof (buffer));
        buffer.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
        buffer.memory = V4L2_MEMORY_MMAP;
        buffer.index = i;

        if (-1 == ioctl (cam_fd, VIDIOC_QUERYBUF, &buffer)) {
            printf ("Video VIDIOC_QUERYBUF failed !\n");
            // return -1;
        }

        //将对应的缓冲块映射到用户空间
        buffers[i].length = buffer.length; /* remember for munmap() */
        buffers[i].start = mmap (NULL, buffer.length,
                            PROT_READ | PROT_WRITE, /* required */
                            MAP_SHARED, /* recommended */
                            cam_fd, buffer.m.offset);


        if (buffers[i].start == MAP_FAILED) {
            printf ("Video mmap failed !\n");
            // return -1;

关于git的常用命令

1 基本linux命令的学习
1.1 cd 改变目录
1.2 cd .. 退回到上一个目录
1.3 pwd 显示当前所在目录
1.4 ls(||) 都是列出当前目录中的所有文件，只不过 || 列出的内容更为详细
1.5 touch 新建一个文件 如 touch index.js 就会在当前目录下 新建一个 index.js文件
1.6 rm 删除一个文件 rm index.js 就会把这个文件删除
1.7 mkdir 新建一个目录 就是新建一个文件夹
1.8 rm -r 删除一个文件夹 rm -r src 删除src目录
1.9 mv 移动文件 mv index.html src index.html 是我们要移动的文件 src 是目标文件夹
1.10 reset 重新初始化终端\清屏
1.11 clear 清屏
1.12 history 查看历史命令
1.13 help 帮助
1.14 exit 退出
1.15 # 表示注释

2 git的核心 （所有理论都源于这个核心）
2.1 工作区域 git本地有三个工作区域 工作目录（working Directory）暂存区（stage/Indx）、资源区（Repositroy或 Git Directroy）如果在加上远程的git仓库（Remote Directroy）就可以分为四个工作区域 四个文件转换如下

workspace：工作区 就是你平时存放项目代码的地方
Index/Stage：暂存区，用于临时存放你改动的地方，事实上它就是个文件，保存即将提交到文佳列表信息
Repositroy：仓库区（或者本地仓库） 就是安全存放数据的位置，这里面有提交到所有版本的数据。其中head指向最新放入仓库的版本
Remote：远程仓库，托管代码的服务器，可以简单认为是项目组中的一台电脑用于远程数据交换 本地的三个区域确切的应该是git仓库中head指向的版本：

3.2 将需要进行版本管理的文件放入到暂存区 git commit
3.3 将暂存区域的文件提交到git仓库 git push
